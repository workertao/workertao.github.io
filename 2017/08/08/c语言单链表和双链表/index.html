单链表

下边是我自己画的单链表的流程图，这里只做了从末尾添加结构体的操作。
单链表之前一直没搞懂是做什么用的，直到有人告诉我这是数据结构之后，我才放心的不去管它，搞java那会一直有人问，你懂数据结构吗？不知道怎么回答，现在看来可以回答我当然知道啦。当然这只是我懒得去搞懂他的借口，我也在百度上搜过大神的回答，链表作为最基本的数据结构，我们势必要去搞懂它。当熟悉了这个流程之后，写起来就简单多了。

typedef struct Node
{
    char *name;        //数据域
    struct Node * next;//指针域   指向下一个节点的地址

}node;

void create(int len){
    node *head ,*p ,*s;
    head = (node*)malloc(sizeof(node*));
    if (head==NULL){
        //没有申请到内存，直接返回
        return;
    }
    head->name = "张三";
    head->next = NULL;
    p = head;
    for (int i = 0; i < len; i++){
        s = (node*)malloc(sizeof(node*));
        if (s == NULL){
            //没有申请到内存，直接返回
            return;
        }
        //拼接字符串
        char cName[20];
        //指针数组，存放要拼接的数据
        char * c[20] = { "a","b","c","d","e","f","g","j","k","l"};
        //char指针，申请内存
        char * cN = (char*)malloc(sizeof(char*));
        if (cN == NULL){
            return;
        }
        //字符串拼接
        strcpy(cName, "李");
        //字符串拼接
        strcat(cName, c[i]);
        //cN指针指向cName
        cN = cName;

        //给当前节点的数据域赋值
        s->name = cN;
        s->next = NULL;

        p->next = s;
        p = s;
        printf("%s   %d\n", p->name, i);
    }
}

int main(){
    create(10);
    system("pause");
    return 0;
}

双链表

双链表的含义其实就是比单链表多了一个指向上一个指针地址的指针域，head结构体的上一个指针域(previous)指向NULL,当前结构体的下一个指针域(next)指向NULL，下边是我理解的流程图。

有了流程图，我们根据图示来撸码。这回我们稍微封装一下，创建一个头文件dlist.h，用来写我们的接口。

#ifndef _DLIST_H
#define _DLIST_H

typedef struct node
{
    int data;
    struct node * previous;
    struct node * next;
} Node;

typedef struct dlist
{
    Node * head;
} Dlist;

void d_init(Dlist * list);//初始化
bool d_insert(Dlist * list, const int data);//插入
bool d_remove(Dlist * list, const int key);//移除
bool d_modify(Dlist * list, const int key, const int data);//修改
Node* d_find(Dlist * list, const int key);//查找
void d_treaverse(Dlist * list, void(*func)(void * pointer));//遍历
void d_datalist(Dlist * list);//遍历
void d_destroy(Dlist * list);//销毁

#endif //_DLIST_H

在.cpp的类里写我们的实现，记得引入头部的依赖

#include "dlist.h"

我这里只写了初始化，插入和遍历，其他的还没写。

static Node* make_node(const int data);//创建节点
static void data_list(Node * current);//遍历链表，打印数据，包括头节点

//@接口
//创建头部 或者  初始化
void d_init(Dlist * list){
    list->head = make_node(-1);
}

//@接口
//参数：list：头结点  data 插入的数据
bool d_insert(Dlist * list, const int data){
    //创建一个Node节点，指向头结点，为了遍历所有节点查找插入位置
    Node * current = list->head;
    //配合data数据域创建一个新的指针节点，便于进行增删改查操作。
    Node * node;
    //从头节点开始遍历，满足条件将当前节点重新赋值。
    while (current->next != NULL && current->data<data){
        current = current->next;
    }
    //如果当前节点的数据跟要插入的数据一样，则返回false
    if (current->data == data){
        printf("添加了重复的数据");
        return false;
    }
    //将数据域传入创建节点的方法，创建一个新的节点，便于上一个节点的指针，进行指向。
    node = make_node(data);
    //如果当前节点是是尾节点，则当前节点指针next==NULL。
    if (current->next == NULL){
        //如果是尾节点，则将指针指向新创建的节点。
        current->next = node;
        //同样，新创建的指针的前驱指向当前的节点
        node->previous = current;
    }
    else{
        //如果当前节点不是尾节点，则当前节点指针next!=NULL,而是指向下一个节点的首地址
        //将新节点的前驱指向当前节点的前驱
        node->previous = current->previous;
        //将新节点的后继指向当前节点，所以这时一个前插方式
        node->next = current;
        //将当前节点的前驱的后继指向新节点
        current->previous->next = node;
        //将当前节点的前驱指向新节点
        current->previous = node;
    }
    return true;
}

//@接口
//遍历
void d_datalist(Dlist * list){
    data_list(list->head->next);

};

//获取当前传入的node节点，打印当前节点的数据域
static void data_list(Node * current){
    Node * a = current;
    while (a != NULL)
    {
        printf("数据：  %d \n", a->data);
        a = a->next;
    }
}

//创建一个新的节点
static Node * make_node(const int data){
    Node * node = (Node *)malloc(sizeof(Node *));
    if (node == NULL)
    {
        exit(1);
        printf("动态分配内存失败，程序结束");
    }
    node->data = data;
    node->previous = NULL;
    node->next = NULL;

    return node;
}

int main(){
    Dlist *d = (Dlist*)malloc(sizeof(Dlist*));
    d_init(d);
    d_insert(d, 90);
    d_insert(d, 6);
    d_insert(d, 50);
    d_insert(d, 18);
    d_insert(d, 20);
    d_insert(d, 30);
    d_datalist(d);
    system("pause");
    return 0;
}

总结：

  1. 不管是单链表还是多链表都包含了数据域和指针域两种结构体变量(结构体内部的变量，我们叫做结构体变量)，而且指针域必须是当前的结构体类型的指针。
  2. 以上事例中，他的数据域是name,指针域是next和previous。
  3. 数据域我们可以自定义，他可以是我们想要的任何数据类型。
  4. 单链表中当前结构体的下一个指针域(next)指向的NULL，上一个结构体的指针域指向的当前的结构体的内存地址。
  5. 双向链表中head结构体的上一个指针域(previous)指向NULL,当前结构体的下一个指针域(next)指向NULL
