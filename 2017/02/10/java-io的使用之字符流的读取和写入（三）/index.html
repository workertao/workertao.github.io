public static void main(String[] args) {
    try {
        // 读取文件对象
        File file = new File("java.txt");
        // 字节流读取
        FileInputStream fis = new FileInputStream(file);
        // 将字节流转换成字符流对象，编码格式默认为GBK，修改为UTF-8
        InputStreamReader isr = new InputStreamReader(fis, "UTF-8");

        // 写入文件对象
        File new_file = new File("java_new.txt");
        // 字节流写入
        FileOutputStream fos = new FileOutputStream(new_file);
        // 将字节流转换成字符流对象，编码格式默认为GBK，修改为UTF-8
        OutputStreamWriter osw = new OutputStreamWriter(fos, "UTF-8");

        // 字符流的读取，我们当然要用字符数组来存储了，同样的字符数组的大小也是因文件大小而定，多次调试，来选择合适的大小可以提高读取和写入效率
        char[] input = new char[100];
        int l = 0;
        //此处用while的意思是因为我们要读取的文件太大，而每回的数组存储有限，所以需要用while循环来多次读取。我们可以把鼠标放到read()的方法上看到他的返回值是一个int，然后后边介绍说，如果读取返回为-1的时候说明读取完成，根据这个我们可以写出下边的代码
        while ((l = isr.read(input)) != -1) {
            //我们为什么要用这个构造呢，因为我们并不清楚要读的字符到底有多少，而每回读取的字符数量又是固定的，为了避免浪费不必要的内存，我们采用以下构造。
            //参数含义1字符数组       2开始写入字符的偏移量      3写入的字符长度
            osw.write(input, 0, l);
        }

        //关闭流，先打开后关闭，后打卡先关闭
        osw.close();
        fos.close();
        isr.close();
        fos.close();

        System.out.println("拷贝完成");
    } catch (FileNotFoundException e) {
        // TODO Auto-generated catch block
        e.printStackTrace();
    } catch (UnsupportedEncodingException e) {
        // TODO Auto-generated catch block
        e.printStackTrace();
    } catch (IOException e) {
        // TODO Auto-generated catch block
        e.printStackTrace();
    }
}
