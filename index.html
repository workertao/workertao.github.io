package gawallet.smart_test.view;

import android.content.Context;
import android.content.res.Resources;
import android.content.res.TypedArray;
import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.graphics.Rect;
import android.util.AttributeSet;
import android.util.DisplayMetrics;
import android.view.View;    
import java.lang.reflect.Array;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.Map;
import gawallet.smart_test.R;


/**
 * Created by yangtao on 2017/3/2.
 */

public class DiyView extends View {
//画笔
private Paint paint;
//x轴单元格
public static int X_ITEM;
//y轴单元格
public static int Y_ITEM;
//Y轴的高度
public static float HEIGHT;

private Context context;

private int windowWidth;
private int windowHeigh;

public DiyView(Context context, AttributeSet attrs) {
    super(context, attrs);
    initAttrs(context, attrs);
}

private void initAttrs(Context context, AttributeSet attrs) {
    this.context = context;
    TypedArray typedArray = context.obtainStyledAttributes(attrs, R.styleable.DiyView);
    X_ITEM = typedArray.getInteger(R.styleable.DiyView_x_item, 0);
    Y_ITEM = typedArray.getInteger(R.styleable.DiyView_y_item, 0);
    HEIGHT = typedArray.getFloat(R.styleable.DiyView_y_height, 0f);
    paint = new Paint();
    typedArray.recycle();
    /**
     * 获取屏幕宽高
     */
    initWindow();
}

private void initWindow() {
    DisplayMetrics displayMetrics = getDisplayMetrics(context);
    windowWidth = displayMetrics.widthPixels;
    windowHeigh = displayMetrics.heightPixels;
}


@Override
protected void onDraw(Canvas canvas) {
    super.onDraw(canvas);
    //设置画笔属性
    paint.setStrokeWidth(10f);
    paint.setColor(Color.BLACK);
    //画横轴
    canvas.drawLine(0, HEIGHT, windowWidth, HEIGHT, paint);
    //画竖轴
    canvas.drawLine(0, 0, 0, HEIGHT, paint);
    //X轴部的段点
    paint.setColor(Color.BLACK);
    for (int i = 0; i < windowWidth / X_ITEM; i++) {
        canvas.drawLine(X_ITEM * i, HEIGHT, X_ITEM * i, HEIGHT - 10, paint);
    }
    //Y轴段点
    for (int i = 0; i < HEIGHT / Y_ITEM; i++) {
        canvas.drawLine(0, HEIGHT - (Y_ITEM * i), 10, HEIGHT - (Y_ITEM * i), paint);
    }
    drawLins(canvas);
}

/**
 * 画折线
 */
private void drawLins(Canvas canvas) {
    ArrayList<Map<String, Float>> drawLins = new ArrayList<Map<String, Float>>();
    Map<String, Float> map = new HashMap<String, Float>();
    map.put("x", 100f);
    map.put("y", 75f);
    Map<String, Float> map1 = new HashMap<String, Float>();
    map1.put("x", 200f);
    map1.put("y", 60f);
    Map<String, Float> map2 = new HashMap<String, Float>();
    map2.put("x", 300f);
    map2.put("y", 208f);
    Map<String, Float> map3 = new HashMap<String, Float>();
    map3.put("x", 400f);
    map3.put("y", 409f);
    Map<String, Float> map4 = new HashMap<String, Float>();
    map4.put("x", 500f);
    map4.put("y", 200f);
    Map<String, Float> map5 = new HashMap<String, Float>();
    map5.put("x", 600f);
    map5.put("y", 498f);
    Map<String, Float> map6 = new HashMap<String, Float>();
    map6.put("x", 700f);
    map6.put("y", 100f);
    Map<String, Float> map7 = new HashMap<String, Float>();
    map7.put("x", 800f);
    map7.put("y", 456f);
    drawLins.add(map);
    drawLins.add(map1);
    drawLins.add(map2);
    drawLins.add(map3);
    drawLins.add(map4);
    drawLins.add(map5);
    drawLins.add(map6);
    drawLins.add(map7);

    //用来保存之前的X，Y坐标
    float oldX = -1f;
    float oldY = -1f;
    //用来保存当前的X，Y坐标
    float x = -1f;
    float y = -1f;
    //先画点
    for (int i = 0; i < drawLins.size(); i++) {
        //x轴坐标不变
        x = drawLins.get(i).get("x");
        //y轴坐标=整体高度减去传入高度
        y = HEIGHT - drawLins.get(i).get("y");
        //先画点
        paint.setColor(Color.BLUE);
        canvas.drawPoint(x, y, paint);
        //画线
        if (oldX != -1f || oldY != -1f) {
            paint.setColor(Color.RED);
            canvas.drawLine(oldX, oldY, x, y, paint);
        }
        //保存当前的X轴Y轴
        oldX = x;
        oldY = y;
    }
}

/**
 * 获取屏幕尺寸与密度.
 *
 * @param context the context
 * @return mDisplayMetrics
 */
public DisplayMetrics getDisplayMetrics(Context context) {
    Resources mResources;
    if (context == null) {
        mResources = Resources.getSystem();
    } else {
        mResources = context.getResources();
    }
    DisplayMetrics mDisplayMetrics = mResources.getDisplayMetrics();
    return mDisplayMetrics;
}
}

下边的代码想稍微啰嗦一下，避免以后自己忘记。我们在自定义view的时候呢，需要重写系统的几个方法，每个构造方法的参数不同，网上有很多关于三个构造函数使用时机的说法,但是说法正确的却没有几家,这里正式的给大家科普一下:(以下赘述摘自网络)

在代码中直接new一个Custom View实例的时候,会调用第一个构造函数.这个没有任何争议.
在xml布局文件中调用Custom View的时候,会调用第二个构造函数.这个也没有争议.
在xml布局文件中调用Custom View,并且Custom View标签中还有自定义属性时,这里调用的还是第二个构造函数.
也就是说,系统默认只会调用Custom View的前两个构造函数,至于第三个构造函数的调用,通常是我们自己在构造函数中主动调用的（例如,在第二个构造函数中调用第三个构造函数）.

至于自定义属性的获取,通常是在构造函数中通过obtainStyledAttributes函数实现的.这里先介绍一下如何生成Custom View的自定义属性.↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓↓

private void initAttrs(Context context, AttributeSet attrs) {
    this.context = context;
    TypedArray typedArray = context.obtainStyledAttributes(attrs, R.styleable.DiyView);
    X_ITEM = typedArray.getInteger(R.styleable.DiyView_x_item, 0);
    Y_ITEM = typedArray.getInteger(R.styleable.DiyView_y_item, 0);
    HEIGHT = typedArray.getFloat(R.styleable.DiyView_y_height, 0f);
    paint = new Paint();
    typedArray.recycle();
    /**
     * 获取屏幕宽高
     */
    initWindow();
}

如上所说我们自定义了标签属性，这些属性的值可以在我们的代码和布局中设置。

<?xml version="1.0" encoding="utf-8"?>
<resources>
<declare-styleable name="DiyView">
<attr name="x_item" format="integer" />
<attr name="y_item" format="integer" />
<attr name="y_height" format="float" />
</declare-styleable>
</resources>

xml布局中声明我们的自定义view和属性并且赋值。

<?xml version="1.0" encoding="utf-8"?>
<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"
xmlns:test="http://schemas.android.com/apk/res-auto"
android:layout_width="match_parent"
android:layout_height="match_parent">

<gawallet.smart_test.view.DiyView
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    test:x_item="100"
    test:y_height="500"
    test:y_item="100" />
</LinearLayout>
